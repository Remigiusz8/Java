    public void actionPerformed(ActionEvent e) {
        if ("cmdGetText".equals(e.getActionCommand())){
            String tekst = textField.getText();
            JOptionPane.showMessageDialog(this, "Odczytany tekst: " + tekst,
            "Informacja!", JOptionPane.INFORMATION_MESSAGE);
        }
        String cmd = e.getActionCommand();
        if ("Zamknij".equals(cmd)) {
            dispose();
        }
        
        Object source = e.getSource();
        if (source == jeden)
            str = str+"1";
        if (source == dwa)
            str = str+"2";
        if (source == trzy)
            str = str+"3";
        if (source == cztery)
            str = str+"4";
        if (source == piec)
            str = str+"5";
        if (source == szesc)
            str = str+"6";
        if (source == siedem)
            str = str+"7";
        if (source == osiem)
            str = str+"8";
        if (source == dziewiec)
            str = str+"9";
        if (source == zero)
            str = str+"0";
                
                
        if (source == plus)
        {
            arg1 = Double.parseDouble(textField.getText());
            str="";
            textField2.setText("");
            dzialanie = 1;
        }
        if (source == minus)
        {
            arg1 = Double.parseDouble(textField.getText());
            str="";
            textField2.setText("");
            dzialanie = 2;
        }
        if (source == gwiazdka)
        {
            arg1 = Double.parseDouble(textField.getText());
            str="";
            textField2.setText("");
            dzialanie = 3;
        }
        if (source == slash)
        {
            arg1 = Double.parseDouble(textField.getText());
            str="";
            textField2.setText("");
            dzialanie = 4;
        }
        if (source == rowna)
        {
            arg2 = Double.parseDouble(textField.getText());
            double wynik = 0;
            switch(dzialanie) {
            case 1:
                wynik = arg1 + arg2;
                break;
            case 2:
                wynik = arg1 - arg2;
                break;
            case 3:
                wynik = arg1 * arg2;
                break;
            case 4:
                if (arg2 == 0) {
                    wynik = 0;
                } else {
                    wynik = arg1 / arg2;
                }
                break;
            default:
                System.out.println("cos sie popsu≈Ço");
                break;
            }
            str = "";
            str = Double.toString(wynik);
            textField2.setText(str);
        }

    
    }
